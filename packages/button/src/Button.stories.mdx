<Meta
  title="Components/Button"
  component={Button}
  argTypes={{
    classModifier: {
      control: {
        type: 'multi-select',
        options: [
          'success',
          'info',
          'danger',

          'hasiconLeft',
          'hasiconRight',
        ],
      },
    },
    icon: {
      description: 'glyphicon name to use in the Action',
      options: ['', 'alert', 'arrow-left'],
      control: { type: 'select' },
    },
    title: {
      control: { type: 'string' },
    },
    onClick: { action: 'onClick' },
  }}
  parameters={{
    layout: 'centered',
  }}
/>

export const Template = ({ text, icon, classModifier, ...args }) => (
  <Button {...args} classModifier={classModifier.join(' ')}>
    <span className="af-btn__text">{text}</span>
    {icon ? <span className={'glyphicon glyphicon-' + icon} /> : ''}
  </Button>
);

## With icons

To add an icon you need to add a `classModifier`, either `hasiconLeft` or `hasiconRight` whether you want the icon on the left of or on the right.

## Circle button with no text

You can make small round buttons, but do not forget to add a `title`, or an `aria-label` to make your button accessible!
